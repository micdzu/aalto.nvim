#!/bin/bash

# Computes factorial recursively.
# Bash doing math is like a fish climbing a tree—possible, but why?
factorial() {
    local n=$1
    if (( n <= 1 )); then
        echo 1
    else
        local prev=$(factorial $((n - 1)))
        echo $(( n * prev ))
    fi
}

n=10
result=$(factorial $n)
echo "factorial($n) = $result"
# 3628800. It worked. Don't push your luck with n=20.
