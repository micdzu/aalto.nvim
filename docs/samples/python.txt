def factorial(n: int) -> int:
    """Return n! using recursion. Python doesn't judge."""
    if n <= 1:
        return 1
    return n * factorial(n - 1)

if __name__ == "__main__":
    value = 10
    print(f"factorial({value}) = {factorial(value)}")
    # 3628800. Elegant, readable, and slower than you'd like.
